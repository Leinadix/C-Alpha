21std::string Preprocessor::processImports(const std::string &source,
22 const std::string &currentFile) {
23 std::istringstream stream(source);
24 std::ostringstream result;
25 std::string line;
26
27 // Get the directory of the current file
28 std::filesystem::path currentFilePath =
29 std::filesystem::absolute(currentFile);
30 std::filesystem::path currentFileDir = currentFilePath.parent_path();
31
32 // Add current file to processed set to prevent circular imports
33 processedFiles.insert(currentFilePath.string());
34
35 while (std::getline(stream, line)) {
36 if (isImportStatement(line)) {
37 std::string importFile = extractFilename(line);
38 std::filesystem::path importPath = currentFileDir / importFile;
39 std::string absolutePath =
40 std::filesystem::absolute(importPath).string();
41
42 // Check for circular imports
43 if (processedFiles.find(absolutePath) != processedFiles.end()) {
44 throw std::runtime_error("Circular import detected: " +
45 importFile);
46 }
47
48 // Read and process the imported file
49 std::string importedCode = readFile(importPath.string());
50 importedCode = processImports(importedCode, absolutePath);
51
52 // Wrap imported code in a namespace
53 result << generateNamespace(importedCode, importFile) << "\n";
54 } else {
55 result << line << "\n";
56 }
57 }
58
59 return result.str();
60}

73std::string Preprocessor::generateNamespace(const std::string &code,
74 const std::string &filename) {
75 std::string ns = sanitizeIdentifier(filename);
76
77 // Convert filename to absolute path
78 std::filesystem::path filePath =
79 std::filesystem::absolute(currentDir / filename);
80 std::string absolutePath = filePath.string();
81
82 std::ostringstream result;
83 result << "// Start of imported file: " << absolutePath << "\n";
84 result << "layout __import_" << ns << " {\n";
85 result << " int _dummy;\n";
86 result << "};\n\n";
87 result << code << "\n";

112std::string Preprocessor::extractFilename(const std::string &importLine) {
113 size_t start = importLine.find('"');
114 if (start == std::string::npos) {
115 throw std::runtime_error(
116 "Invalid import statement: missing opening quote");
117 }
118
119 size_t end = importLine.find('"', start + 1);
120 if (end == std::string::npos) {
121 throw std::runtime_error(
122 "Invalid import statement: missing closing quote");
123 }
124
125 return importLine.substr(start + 1, end - start - 1);
126}
