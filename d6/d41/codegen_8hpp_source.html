37// Memory management for variables and layout members with proper scoping
38class MemoryManager {
39 private:
40 int nextMemoryAddress{1};
41
42 // Scope stack for variable management using FQDNs
43 std::vector<std::unordered_map<std::string, int>> scopeStack;
44
45 // Layout member offsets (global)
46 std::unordered_map<std::string, std::unordered_map<std::string, int>>
47 layoutMemberOffsets;
48
49 // Track current scope path for FQDN construction
50 std::vector<std::string> currentScopePath;
51
52 [[nodiscard]] std::string getCurrentFQDNPrefix() const {
53 if (currentScopePath.empty())
54 return "global";
55 std::string prefix;
56 for (const auto &scope : currentScopePath) {
57 if (!prefix.empty())
58 prefix += "::";
59 prefix += scope;
60 }
61 return prefix;
62 }
63
64 public:
65 MemoryManager() {
66 // Start at address 1 (0 might be reserved)
67 // Create global scope
68 scopeStack.emplace_back();
69 currentScopePath.emplace_back("global");
70 }
71
72 [[nodiscard]] int getNextMemoryAddress() const {
73 return nextMemoryAddress;
74 }
75
76 // Scope management
77 void pushScope(const std::string &scopeName) {
78 scopeStack.emplace_back();
79 currentScopePath.push_back(scopeName);
80 }
81
82 void popScope() {
83 if (scopeStack.size() <= 1) {
84 throw std::runtime_error("Cannot pop global scope");
85 }
86 scopeStack.pop_back();
87 currentScopePath.pop_back();
88 }
89
90 // Variable memory management with FQDN support
91 int allocateMemory(const std::string &fqdn, int size = 1) {
92 if (hasVariableInCurrentScope(fqdn)) {
93 throw std::runtime_error(
94 "Variable '" + fqdn +
95 "' already has memory allocated in current scope");
96 }
97
98 const int address = nextMemoryAddress;
99 scopeStack.back()[fqdn] = address;
100 nextMemoryAddress += size;
101 return address;
102 }
103
104 int allocateArray(const int size) {
105 const int address = nextMemoryAddress;
106 nextMemoryAddress += size;
107 return address;
108 }
109
110 [[nodiscard]] int getVariableAddress(const std::string &fqdn) const {
111 // Search from current scope upwards
112 for (const auto &it : std::ranges::reverse_view(scopeStack)) {
113 if (auto varIt = it.find(fqdn); varIt != it.end()) {
114 return varIt->second;
115 }
116 }
117
118 throw std::runtime_error("Variable '" + fqdn + "' not found in memory");
119 }
120
121 [[nodiscard]] bool hasVariable(const std::string &fqdn) const {
122 // Search from current scope upwards
123 for (const auto &it : std::ranges::reverse_view(scopeStack)) {
124 if (it.contains(fqdn)) {
125 return true;
126 }
127 }
128 return false;
129 }
130
131 [[nodiscard]] bool hasVariableInCurrentScope(const std::string &fqdn) const {
132 if (scopeStack.empty()) {
133 return false;
134 }
135 return scopeStack.back().contains(fqdn);
136 }
137
138 // Layout management with FQDN support
139 void setLayoutMemberOffset(const std::string &layoutFQDN,
140 const std::string &memberName, int offset) {
141 layoutMemberOffsets[layoutFQDN][memberName] = offset;
142 }
143
144 [[nodiscard]] int getLayoutMemberOffset(const std::string &layoutFQDN,
145 const std::string &memberName) const {
146 if (const auto layoutIt = layoutMemberOffsets.find(layoutFQDN); layoutIt != layoutMemberOffsets.end()) {
147 if (const auto memberIt = layoutIt->second.find(memberName); memberIt != layoutIt->second.end()) {
148 return memberIt->second;
149 }
150 }
151 throw std::runtime_error("Member '" + memberName +
152 "' not found in layout '" + layoutFQDN + "'");
153 }
154
155 void clearAll() {
156 nextMemoryAddress = 1;
157 scopeStack.clear();
158 layoutMemberOffsets.clear();
159 currentScopePath.clear();
160 // Reinitialize global scope
161 scopeStack.emplace_back();
162 currentScopePath.emplace_back("global");
163 }
